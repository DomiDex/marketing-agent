## Keyword Extraction Script

To automatically extract keywords from SKILL.md frontmatter:

```typescript
// scripts/extract-keywords.ts

import { readdir, readFile } from 'fs/promises'
import { parse } from 'yaml'

async function extractKeywords() {
  const skillsDir = '../.claude/skills'
  const skills = await readdir(skillsDir)

  const results: Record<string, string[]> = {}

  for (const skill of skills) {
    const skillPath = `${skillsDir}/${skill}/SKILL.md`
    const content = await readFile(skillPath, 'utf-8')

    // Extract frontmatter
    const frontmatterMatch = content.match(/^---\n([\s\S]*?)\n---/)
    if (!frontmatterMatch) continue

    const frontmatter = parse(frontmatterMatch[1])

    // Extract keywords from description
    const description = frontmatter.description || ''
    const keywords = extractKeywordsFromDescription(description)

    results[skill] = keywords
  }

  return results
}

function extractKeywordsFromDescription(description: string): string[] {
  const keywords: string[] = []

  // Extract quoted phrases
  const quoted = description.match(/"([^"]+)"/g)
  if (quoted) {
    keywords.push(...quoted.map(q => q.replace(/"/g, '')))
  }

  // Extract "when the user mentions" patterns
  const mentionMatch = description.match(/when the user mentions\s+(.+?)(?:\.|$)/i)
  if (mentionMatch) {
    const mentions = mentionMatch[1].split(/,\s*(?:or\s+)?/)
    keywords.push(...mentions.map(m => m.replace(/"/g, '').trim()))
  }

  return [...new Set(keywords)] // Dedupe
}
```
